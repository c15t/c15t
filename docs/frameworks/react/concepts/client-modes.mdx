export function ConsentManager() {
  return (
    <ConsentManagerProvider
      options={{
        mode: 'c15t',

<import src="../../../shared/concepts/client-modes.mdx#custom-mode" />

```tsx
import { ConsentManagerProvider, ConsentBanner } from '@c15t/react';

export function ConsentManager() {
  return (
    <ConsentManagerProvider
      options={{
        mode: 'custom',
        endpointHandlers: {
          async fetchConsent() {
            const res = await fetch('/my-api/consent');
            const data = await res.json();
            return {
              ok: true,
              data: {
                jurisdiction: data.jurisdiction,
                consents: data.preferences,
              },
            };
          },
          async setConsent(body) {
            await fetch('/my-api/consent', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify(body),
            });
            return { ok: true, data: null };
          },
        },
      }}
    >
      <ConsentBanner />
      {children}
    </ConsentManagerProvider>
  );
}
```
