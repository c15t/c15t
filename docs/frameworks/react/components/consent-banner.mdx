## Basic Usage

```tsx
import { ConsentManagerProvider, ConsentBanner } from '@c15t/react';

export function ConsentManager() {
  return (
    <ConsentManagerProvider options={{ mode: 'hosted', backendURL: 'https://your-instance.c15t.dev' }}>
      <ConsentBanner />
    </ConsentManagerProvider>
  );
}
```
