## React Example

```tsx
import type { ReactNode } from 'react';
import {
  ConsentBanner,
  ConsentDialog,
  ConsentManagerProvider,
  policyPackPresets,
} from '@c15t/react';

export function ConsentManager({ children }: { children: ReactNode }) {
  return (
    <ConsentManagerProvider
      options={{
        mode: 'offline',
        policyPacks: [
          policyPackPresets.europeOptIn(),
          policyPackPresets.californiaOptOut(),
          policyPackPresets.worldNoBanner(),
        ],
        overrides: {
          country: 'DE', // Preview as a German visitor
        },
      }}
    >
      <ConsentBanner />
      <ConsentDialog />
      {children}
    </ConsentManagerProvider>
  );
}
```

When your app uses a live backend, move the canonical pack into backend `policyPacks` and keep frontend `policyPacks` only for local previews, stories, or tests.
