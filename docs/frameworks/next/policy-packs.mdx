## Next.js Example

```tsx title="components/consent-manager/provider.tsx"
'use client';

import type { ReactNode } from 'react';
import {
  ConsentBanner,
  ConsentDialog,
  ConsentManagerProvider,
  policyPackPresets,
} from '@c15t/nextjs';

export function ConsentManager({ children }: { children: ReactNode }) {
  return (
    <ConsentManagerProvider
      options={{
        mode: 'offline',
        policyPacks: [
          policyPackPresets.europeOptIn(),
          policyPackPresets.californiaOptOut(),
          policyPackPresets.worldNoBanner(),
        ],
        overrides: {
          country: 'DE', // Preview as a German visitor
        },
      }}
    >
      <ConsentBanner />
      <ConsentDialog />
      {children}
    </ConsentManagerProvider>
  );
}
```

For production Next.js apps, prefer a backend route backed by `policyPacks` and optionally hydrate the first `/init` response with the [server-side utilities](/docs/frameworks/next/server-side).
