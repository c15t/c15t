## Basic Usage

```tsx
import { type ReactNode } from 'react';
import {
  ConsentManagerProvider,
  IABConsentBanner,
  IABConsentDialog,
} from '@c15t/nextjs';

export default function ConsentManager({ children }: { children: ReactNode }) {
  return (
    <ConsentManagerProvider
      options={{
        mode: 'c15t',
        backendURL: '/api/c15t',
        iab: {
          enabled: true,
          cmpId: 123,
          vendors: [1, 2, 10, 25],
        },
      }}
    >
      <IABConsentBanner />
      <IABConsentDialog />
      {children}
    </ConsentManagerProvider>
  );
}
```
