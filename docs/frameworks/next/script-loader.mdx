## Implementation

<Tabs items={['App Directory', 'Pages Directory']}>
  <Tab value="App Directory">
    Due to the nature of Next.js App Directory, we have to define callbacks to run in a client component.

    ```ts title="app/layout.client.tsx"
    'use client';

    import { ClientSideOptionsProvider } from '@c15t/nextjs/client';
    import { googleTagManager } from '@c15t/nextjs/scripts/google-tag-manager';

    export function ClientProvider() {
    	return (
    		<ClientSideOptionsProvider
    			scripts={[
    				googleTagManager({
    					id: 'GTM-XXXXXXX',
    				}), // Prebuilt script or...
    				{
    					id: 'example',
    					src: 'https://analytics.example.com/script.js',
    					category: 'analytics',
    				}
    			]}
    		/>
    	);
    }
    ```

    <Callout type="note">
      Ensure you import this component inside the ConsentManagerProvider component in your `app/layout.tsx` file.
    </Callout>

    ```ts title="app/layout.tsx"
    import { ConsentManagerProvider } from '@c15t/nextjs';
    import { ClientProvider } from './layout.client';
    import type { ReactNode } from 'react';

    export default function Layout({ children }: { children: ReactNode }) {
      return (
        <ConsentManagerProvider options={{
          // your options here
        }}>
          <ClientProvider />
          {children}
        </ConsentManagerProvider>
      );
    }
    ```
  </Tab>

  <Tab value="Pages Directory">
    ```ts title="pages/_app.tsx"
    import { ConsentManagerProvider } from '@c15t/nextjs/pages';
    import { googleTagManager } from '@c15t/nextjs/scripts/google-tag-manager';
    import type { AppProps } from 'next/app';

    export default function App({ Component, pageProps }: AppProps) {
    return (
      <ConsentManagerProvider options={{
        scripts: [
          googleTagManager({
            id: 'GTM-XXXXXXX',
          }), // Prebuilt script or...
          {
            id: 'example',
            src: 'https://analytics.example.com/script.js',
            category: 'analytics',
          }
        ]
      }}>
        <Component {...pageProps} />
      </ConsentManagerProvider>
    );
    }
    ```
  </Tab>
</Tabs>
