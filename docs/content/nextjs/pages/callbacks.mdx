## Available callbacks

### onBannerFetched

This callback is called when the consent banner is fetched, it is not called when the banner is set to `offline` mode.

### onConsentSet

This callback is called when the consent is set.

<auto-type-table path="../../../node_modules/c15t/src/types/callbacks.ts" name="OnConsentSetPayload" />

### onError

This callback is called when an error occurs.

## Adding Callbacks
```tsx title="pages/_app.tsx"
import { 
  ConsentManagerDialog,
  ConsentManagerProvider,
  CookieBanner,
} from '@c15t/nextjs/pages';

export default function App({ Component, pageProps }: AppProps) {
	return (
		<ConsentManagerProvider
			initialData={pageProps.initialC15TData}
			options={{
				mode: 'c15t',
				backendURL: '/api/c15t',
				consentCategories: ['necessary', 'marketing'],
				ignoreGeoLocation: true,
				callbacks: {
					onBannerFetched(response) {
						console.log('Consent banner fetched', response);
					},
					onConsentSet(response) {
						console.log('Consent has been saved', response);
					},
					onError(response) {
						console.log('Error', response);
					},
				},
			}}
		>
			<CookieBanner />
			<ConsentManagerDialog />
			<Component {...pageProps} />
		</ConsentManagerProvider>
	);
}
```
